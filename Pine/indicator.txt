//@version=5
indicator("CohesivX BTC Fair Zone — Manual Snapshot", overlay=true, max_labels_count=50)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// INPUTURI PRINCIPALE
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

centralPrice = input.float(77070, "Cohesive central price", step=100)
lowerBand    = input.float(62000, "Lower structural band", step=100)
upperBand    = input.float(95000, "Upper structural band", step=100)

efficientCost   = input.float(45000, "Efficient miner cost", step=100)
standardCost    = input.float(65000, "Standard miner cost", step=100)
inefficientCost = input.float(85000, "Inefficient miner cost", step=100)

showBand        = input.bool(true, "Show cohesive band")
showEnergyLines = input.bool(true, "Show miner cost lines")
showTable       = input.bool(true, "Show status table")
showBackground  = input.bool(true, "Highlight extreme deviation")

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// CALCULE
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

price = close

deviationPct = centralPrice > 0 ? ((price - centralPrice) / centralPrice) * 100 : na
absDeviation = math.abs(deviationPct)

zoneStatus =
     absDeviation < 8  ? "NORMAL" :
     absDeviation < 18 ? "STRONG DEVIATION" :
                         "EXTREME DEVIATION"

zoneBias =
     price > centralPrice ? "ABOVE cohesive price" :
     price < centralPrice ? "BELOW cohesive price" :
                            "AT cohesive price"

productionRatio = standardCost > 0 ? price / standardCost : na

minerStatus =
     productionRatio >= 1.50 ? "COMFORTABLE" :
     productionRatio >= 1.15 ? "PROFITABLE" :
     productionRatio >= 1.00 ? "AT LIMIT" :
     productionRatio >= 0.85 ? "BELOW COST" :
                               "CAPITULATION"

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// CULORI LOGICE
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

centralColor = color.rgb(255, 204, 0)
upperColor   = color.rgb(255, 120, 120)
lowerColor   = color.rgb(80, 180, 255)
bandColor    = color.new(color.rgb(120, 180, 255), 88)

efficientColor   = color.rgb(0, 200, 120)
standardColor    = color.rgb(255, 180, 0)
inefficientColor = color.rgb(255, 80, 80)

extremeBg =
     absDeviation >= 18 ? color.new(color.red, 88) :
     absDeviation >= 8  ? color.new(color.orange, 90) :
                          na

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// PLOTARE ZONĂ COEZIVĂ
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

pUpper = plot(showBand ? upperBand : na, "Upper structural band", color=upperColor, linewidth=1)
pLower = plot(showBand ? lowerBand : na, "Lower structural band", color=lowerColor, linewidth=1)

fill(pUpper, pLower, color=showBand ? bandColor : na, title="Cohesive fair zone")

plot(centralPrice, "Cohesive central price", color=centralColor, linewidth=3)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// LINII COST PRODUCȚIE
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

plot(showEnergyLines ? efficientCost : na, "Efficient miner cost", color=efficientColor, linewidth=1)
plot(showEnergyLines ? standardCost : na, "Standard miner cost", color=standardColor, linewidth=2)
plot(showEnergyLines ? inefficientCost : na, "Inefficient miner cost", color=inefficientColor, linewidth=1)

bgcolor(showBackground ? extremeBg : na)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// TABEL STATUS
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

var table t = table.new(position.top_right, 2, 7, border_width=1)

if barstate.islast and showTable
    table.cell(t, 0, 0, "CohesivX BTC", text_color=color.white, bgcolor=color.rgb(30, 30, 30))
    table.cell(t, 1, 0, "Fair Zone", text_color=color.white, bgcolor=color.rgb(30, 30, 30))

    table.cell(t, 0, 1, "Live price", text_color=color.white)
    table.cell(t, 1, 1, str.tostring(math.round(price)), text_color=color.white)

    table.cell(t, 0, 2, "Central", text_color=color.white)
    table.cell(t, 1, 2, str.tostring(math.round(centralPrice)), text_color=centralColor)

    table.cell(t, 0, 3, "Deviation", text_color=color.white)
    table.cell(t, 1, 3, str.tostring(deviationPct, "#.##") + "%", text_color=color.white)

    table.cell(t, 0, 4, "Zone status", text_color=color.white)
    table.cell(t, 1, 4, zoneStatus, text_color=color.white)

    table.cell(t, 0, 5, "Position", text_color=color.white)
    table.cell(t, 1, 5, zoneBias, text_color=color.white)

    table.cell(t, 0, 6, "Miner status", text_color=color.white)
    table.cell(t, 1, 6, minerStatus, text_color=color.white)

//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
// ETICHETĂ PE ULTIMA LUMÂNARE
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

var label lastLabel = na

if barstate.islast
    label.delete(lastLabel)

    labelText =
         "CohesivX BTC Fair Zone\n" +
         "Price: " + str.tostring(math.round(price)) + "\n" +
         "Central: " + str.tostring(math.round(centralPrice)) + "\n" +
         "Deviation: " + str.tostring(deviationPct, "#.##") + "%\n" +
         zoneStatus + "\n" +
         minerStatus

    lastLabel := label.new(
         bar_index,
         price,
         labelText,
         style=label.style_label_left,
         textcolor=color.white,
         color=color.rgb(20, 20, 20)
    )
