f_interp(_t1, _v1, _t2, _v2) =>
    span = float(_t2 - _t1)
    pos = span != 0 ? float(time - _t1) / span : 0.0
    posClamped = math.max(0.0, math.min(1.0, pos))
    _v1 + (_v2 - _v1) * posClamped

f_year_value(_v2013, _v2014, _v2015, _v2016, _v2017, _v2018, _v2019, _v2020, _v2021, _v2022, _v2023, _v2024, _v2025, _v2026) =>
    float out = na
    if time < t2014
        out := f_interp(t2013, _v2013, t2014, _v2014)
    else if time < t2015
        out := f_interp(t2014, _v2014, t2015, _v2015)
    else if time < t2016
        out := f_interp(t2015, _v2015, t2016, _v2016)
    else if time < t2017
        out := f_interp(t2016, _v2016, t2017, _v2017)
    else if time < t2018
        out := f_interp(t2017, _v2017, t2018, _v2018)
    else if time < t2019
        out := f_interp(t2018, _v2018, t2019, _v2019)
    else if time < t2020
        out := f_interp(t2019, _v2019, t2020, _v2020)
    else if time < t2021
        out := f_interp(t2020, _v2020, t2021, _v2021)
    else if time < t2022
        out := f_interp(t2021, _v2021, t2022, _v2022)
    else if time < t2023
        out := f_interp(t2022, _v2022, t2023, _v2023)
    else if time < t2024
        out := f_interp(t2023, _v2023, t2024, _v2024)
    else if time < t2025
        out := f_interp(t2024, _v2024, t2025, _v2025)
    else if time < t2026
        out := f_interp(t2025, _v2025, t2026, _v2026)
    else
        out := f_interp(t2026, _v2026, t2027, _v2026)
    out
