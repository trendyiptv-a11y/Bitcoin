f_interp(_t1, _v1, _t2, _v2) =>
    span = float(_t2 - _t1)
    pos = span != 0 ? float(time - _t1) / span : 0.0
    posClamped = math.max(0.0, math.min(1.0, pos))
    _v1 + (_v2 - _v1) * posClamped

f_year_value(_v2013, _v2014, _v2015, _v2016, _v2017, _v2018, _v2019, _v2020, _v2021, _v2022, _v2023, _v2024, _v2025, _v2026) =>
    float out = na
    if time < t2014
        out := f_interp(t2013, _v2013, t2014, _v2014)
    else if time < t2015
        out := f_interp(t2014, _v2014, t2015, _v2015)
    else if time < t2016
        out := f_interp(t2015, _v2015, t2016, _v2016)
    else if time < t2017
        out := f_interp(t2016, _v2016, t2017, _v2017)
    else if time < t2018
        out := f_interp(t2017, _v2017, t2018, _v2018)
    else if time < t2019
        out := f_interp(t2018, _v2018, t2019, _v2019)
    else if time < t2020
        out := f_interp(t2019, _v2019, t2020, _v2020)
    else if time < t2021
        out := f_interp(t2020, _v2020, t2021, _v2021)

        out := f_interp(t2021, _v2021, t2022, _v2022)

    else if time < t2024

        out := f_interp(t2025, _v2025, t2026, _v2026)

    out

f_pos(_p) =>
    pos = position.top_right
    if _p == "Top left"
        pos := position.top_left
    else if _p == "Bottom right"
        pos := position.bottom_right
    else if _p == "Bottom left"
        pos := position.bottom_left
    pos

f_row(_table, _row, _label, _value, _valueColor) =>
    table.cell(_table, 0, _row, _label, text_color=labelColor, bgcolor=bgColor)
    table.cell(_table, 1, _row, _value, text_color=_valueColor, bgcolor=bgColor)

f_blank(_table, _row) =>
    table.cell(_table, 0, _row, "", text_color=color.new(color.white, 100), bgcolor=color.new(color.black, 100))
    table.cell(_table, 1, _row, "", text_color=color.new(color.white, 100), bgcolor=color.new(color.black, 100))
